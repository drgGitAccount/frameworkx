Modal

Use VisualUI's JavaScript modal plugin to add dialogs to your site for lightboxes, user notifications, or completely custom content.

Live Demo

Modal title

Modal body text goes here.

Code

<!-- Button trigger -->
<button type="button" class="fw-btn fw-btn-primary" data-toggle="modal" data-target="#exampleModal">
    Launch demo modal
</button>

<!-- Modal -->
<div class="fw-modal fade" id="exampleModal" tabindex="-1">
    <div class="fw-modal-dialog">
        <div class="fw-modal-content">
            <div class="fw-modal-header">
                <h5 class="fw-modal-title">Modal title</h5>
                <button type="button" class="fw-btn-close" data-dismiss="modal"></button>
            </div>
            <div class="fw-modal-body">
                <p>Modal body text goes here.</p>
            </div>
            <div class="fw-modal-footer">
                <button type="button" class="fw-btn fw-btn-secondary" data-dismiss="modal">Close</button>
                <button type="button" class="fw-btn fw-btn-primary">Save changes</button>
            </div>
        </div>
    </div>
</div>
